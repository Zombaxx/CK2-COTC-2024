join_from_marxist_society_society = { #join the society of someone
	FROM = {
		if = {
			limit = {
				society_member_of = marxist_society
			}
			ROOT = {
				join_society = marxist_society
			}
		}
	}
}

detract_marxist_society_currency_preach_effect = {
	if = {
		limit = { is_in_society = yes }
		change_society_currency = -500
	}
}
