				OR = {
					government = military_directorate

				liege = { character = ROOT }
			}
			opinion = { who = ROOT modifier = opinion_granted_me_claim }
			hidden_tooltip = { character_event = { id = military.301 } }

			limit = { liege_before_war = { character = ROOT } }
			opinion = { who = ROOT modifier = opinion_granted_claim_on_my_title }
			hidden_tooltip = {
				character_event = { id = military.302 }
			}
		}
	}
	option = {
		name = "EVTOPTNEVERMIND"
		clr_character_flag = flag_authorizing_occupations
		ai_chance = { factor = 0 }
	}

# County Conquest
character_event = {
	id = military.2000
	hide_window = yes

	min_age = 20
	only_rulers = yes
	only_capable = yes
	prisoner = no

	trigger = {
		ai = yes
		independent = yes
		has_regent = no
		war = no
		NOT = { has_character_modifier = constitutional_crisis }
		wealth = 0
		monthly_income = 0
		OR = {
			religion = terran_imperial
			religion = astrist
			religion = neo_feudal
		}
		any_realm_province = {
			any_neighbor_province = {
				county = {
					holder_scope = {
						top_liege = {
							NOT = { same_realm = ROOT }
							NOT = { reverse_opinion = { who = ROOT value = 0 } }
							NOT = { has_truce = ROOT }
							NOT = { relative_power = { power = 1.0 who = ROOT } }
							NOT = {
								any_allied_character = {
									OR = {
										character = ROOT
										relative_power = { power = 1.0 who = ROOT }
									}
								}
							}
						}
					}
				}
			}
		}
	}
	mean_time_to_happen = {
		months = 120
		modifier = {
			factor = 0.25
			has_ambition = obj_become_emperor
		}
		modifier = {
			factor = 0.5
			trait = ambitious
		}
		modifier = {
			factor = 0.5
			trait = cold
		}
		modifier = {
			factor = 2
			trait = content
		}
		modifier = {
			factor = 2
			trait = friendly
		}
		modifier = {
			factor = 2
			trait = craven
		}
	}
	immediate = {
		random_realm_province = {
			limit = {
				any_neighbor_province = {
					county = {
						holder_scope = {
							top_liege = {

								NOT = { relative_power = { power = 1.0 who = ROOT } }
								NOT = {
									any_allied_character = {
										OR = {
											character = ROOT
											relative_power = { power = 1.0 who = ROOT }
										}
									}
								}
							}
						}
					}
				}
			}
			random_neighbor_province = {
				limit = {
					county = {
						holder_scope = {
							top_liege = {
								NOT = { same_realm = ROOT }
								NOT = { reverse_opinion = { who = ROOT value = 0 } }
								NOT = { has_truce = ROOT }
								NOT = { relative_power = { power = 1.0 who = ROOT } }
								NOT = {
									any_allied_character = {

	id = military.3000
	hide_window = yes

	only_rulers = yes
	min_age = 18
	prisoner = no

		any_demesne_title = {
			is_tribal_type_title = yes
			dejure_liege_title = { holder_scope = { is_liege_of = ROOT } }
		}
		liege = {
			ai = yes
			any_vassal = {
				OR = {
					government = military_directorate

				}
