			limit = {
				NOT = { tier = king }

				kingdom = {
					holder_scope = {
						set_defacto_vassal = PREVPREVPREV

			}
		}
		any_vassal = {
			limit = {
				NOT = { tier = king }

						holder_scope = {
							NOT = { same_liege = PREVPREVPREV }

						}
					}
				}
			}
			set_defacto_liege = THIS

			any_demesne_title = {
				limit = { tier = emperor }

		name = "EVTOPTNEVERMIND"

	}
