#An envoy from Mexikha appears, hidden event generates the envoy then pings
character_event = {
	id = SSI.60000
	hide_window = yes
	is_triggered_only = yes

    only_capable = yes
    only_playable = yes
    has_dlc = "Sunset Invasion"
    prisoner = no
    war = no
    min_age = 16

    trigger = {
        NOT = { offmap_mexikha = { governor = { character = ROOT } } }
        is_inaccessible_trigger = no
        mexikha_diplo_interaction_trigger = yes
        offmap_mexikha = {
            NOR = {
                has_status = mexikha_civil_war
                has_status = mexikha_plague
            }
        }
    }
	
	weight_multiplier = {
		days = 1
		modifier = {
			factor = 0.1
			any_artifact = { has_artifact_flag = mexikhan }
		}
        modifier = {
            offmap_mexikha = { has_status = mexikha_unrest }
            factor = 0.6
        }
        modifier = {
            offmap_mexikha = { has_status = mexikha_norse_invasion }
            factor = 0.3
        }
        modifier = {
            offmap_mexikha = { has_status = mexikha_inca_invasion }
            factor = 0.3
        }
        modifier = {
            offmap_mexikha = { has_status = mexikha_famine }
            factor = 0.3
        }
        modifier = {
            offmap_mexikha = { has_status = mexikha_golden_age }
            factor = 3
        }
        modifier = {
            offmap_mexikha = { has_policy = mexikha_expansionist }
            factor = 2
        }
	}
	
	immediate = {
		offmap_mexikha = {
			governor = {
				random_list = {
					50 = {
						modifier = {
							NOT = {
								has_game_rule = {
									name = gender
									value = all
								}
							}
							factor = 0.05
						}
						create_character = {
							age = 32
							culture = PREV
							religion = PREV
							dynasty = actually_culture
							female = yes
							random_traits = yes
               				historical = yes
						}
					}
					50 = {
						create_character = {
							age = 32
							culture = PREV
							religion = PREV
							dynasty = actually_culture
							female = no
							random_traits = yes
							historical = yes
						}
					}
				}
				new_character = {
					set_character_flag = mexikhan_envoy_artifacts
					random = {
						chance = 10
						remove_education_effect = yes
						add_trait = brilliant_strategist
						change_martial = 5
						set_character_flag = mexikhan_envoy_warrior
					}
					character_event = { id = SSI.60001 }
				}
			}
		}
	}
}
