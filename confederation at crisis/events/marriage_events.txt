# Give characters a chance to arrange romantic marriages instead of political ones
character_event = {
	id = marriage.0
	hide_window = yes
	
	min_age = 18
	max_age = 40
	only_capable = yes
	prisoner = no
	is_triggered_only = yes
	
	trigger = {
		is_married = no
		is_betrothed = no
		NOT = { trait = not_ready_to_marry }
		NOT = { trait = hiver_implant }
		NOT = { has_character_flag = flag_ready_to_marry }
		
		# Some ideologies don't put up with this nonsense
		OR = {
			AND = {
				NOT = { religion = terran_imperial }
				NOT = { religion = neo_feudal }
				NOT = { religion_group = religious_group }
				NOT = { religion = machine_cultist }
				NOT = { religion = bio_directionist }
			}
			trait = carefree
			trait = pragmatic
		}
	}
	
	immediate = { add_trait = not_ready_to_marry }
}

# Patrician dynasty in imminent danger due to lack of appropriate marriages
# Female version
character_event = {
	id = marriage.50
	hide_window = yes
	
	only_women = yes
	min_age = 16
	max_age = 50
	prisoner = no
	capable_only = yes
	
	trigger = {
		is_landed = no
		liege = {
			dynasty = ROOT
			is_patrician = yes
			any_courtier = {
				is_landed = no
				NOT = { any_heir_title = { always = yes } }
				NOT = { any_pretender_title = { always = yes } }
				is_female = no
				is_married = no
				fertility = 0.1
				NOT = { num_of_lovers = 1 }
				NOT = { age = 50 }
				NOT = { is_close_relative = ROOT }
				NOT = { dynasty = ROOT }
				age = 16
			}
		}
		is_married = no
		NOT = { num_of_lovers = 1 }
		fertility = 0.1
		NOT = {
			any_dynasty_member = {
				NOT = { age = 50 }
				OR = {
					AND = {
						is_female = no
						is_married_matrilineally = no
					}
					AND = {
						is_female = yes
						is_married_matrilineally = yes
					}
				}
				NOT = { trait = bastard }
				NOT = { trait = incapable }
				fertility = 0.1
			}
		}
	}
	
	mean_time_to_happen = {
		months = 24
		
		modifier = {
			factor = 10
			has_character_flag = flag_ignore_patrician_matrilineal_marriage_request
		}
	}
	
	immediate = {
		liege = {
			random_courtier = {
				limit = {
					is_landed = no
					NOT = { any_heir_title = { always = yes } }
					NOT = { any_pretender_title = { always = yes } }
					is_female = no
					is_married = no
					fertility = 0.1
					NOT = { num_of_lovers = 1 }
					NOT = { age = 50 }
					NOT = { is_close_relative = ROOT }
					NOT = { dynasty = ROOT }
					age = 16
				}
				character_event = { id = marriage.51 }
			}
		}
	}
}
