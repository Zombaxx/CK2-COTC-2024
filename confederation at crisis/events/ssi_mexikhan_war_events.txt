character_event = {
	id = SSI.30040
	
	hide_window = yes

	has_dlc = "Sunset Invasion"

	is_triggered_only = yes
	
	immediate = {
		offmap_mexikha = {
			if = {
				limit = {
					NOT = { has_offmap_flag = mexikha_entered_war }
				}
				set_offmap_flag = mexikha_entered_war
			}
		}

		if = { #Variables set elsewhere, run effects and notification event
			limit = { NOT = { has_offmap_tmp_flag = mexikha_war_variables_set } }

			### To manipulate the troop spawning for specific situations modify the parameter variables (mexikha_war_troop_quantity, mexikha_war_troop_quality, mexikha_war_target_match_mult, and mexikha_war_minimum_commander_martial) after the following block.
			set_variable = {
				which = mexikha_war_troop_quantity
				value = 16
				# 16 regiments of ~500 units
			}
			set_variable = {
				which = mexikha_war_target_match_mult
				value = 0.95
			}
			set_variable = {
				which = mexikha_war_troop_quality
				value = 3
				# Value between 1-7
			}
			set_variable = {
				which = mexikha_war_minimum_commander_martial
				value = 12
			}
			###
		}
		else = { clr_offmap_tmp_flag = mexikha_war_variables_set }

		# Manipulate the parameter variables dependent on the current status in Mexikha.		
		mexikha_war_modify_variables_by_status_effect = yes
		# Manipulate the parameter variables dependent on the current century.
		mexikha_war_modify_variables_by_century_effect = yes
		# Manipulate the parameter variables dependent on the current realm size.
		mexikha_war_modify_variables_by_realm_size = yes
		# Manipulate the parameter variables dependent on defensive war against revolts
		mexikha_war_modify_variables_by_being_defensive = yes
		
		### Match mult calculation against the opponents troops.
		export_to_variable = {
			which = mexikha_temp_troop_calc
			value = realm_levies_plus_allies
			who = FROM
		}
		
		divide_variable = {
			which = mexikha_temp_troop_calc
			value = 500
		}
		
		multiply_variable = {
			which = mexikha_temp_troop_calc
			which = mexikha_war_target_match_mult
		}
		
		change_variable = {
			which = mexikha_war_troop_quantity
			which = mexikha_temp_troop_calc
		}
		###
		
		### Remove troops for a portion of levies of the mexikhan protectorate.
		export_to_variable = {
			which = mexikha_temp_troop_calc
			value = realm_levies
		}
		
		subtract_variable = {
			which = mexikha_temp_troop_calc
			value = 7500 ## AI extra consideration from the spawned troops needs to be removed here
		}
		
		divide_variable = {
			which = mexikha_temp_troop_calc
			value = 780 # For every 780 troops of levies Mainland will send 500 troops less to help.
		}
		
		subtract_variable = {
			which = mexikha_war_troop_quantity
			which = mexikha_temp_troop_calc
		}
		###
		
		any_courtier_or_vassal = { # Copy the variable to all courtiers and vassals
			set_variable = {
				which = mexikha_war_minimum_commander_martial
				which = PREV
			}
		}
		
		if = {
			limit = {
				NOT = {
					any_courtier_or_vassal = {
						#count = 3
						count = 2
						martial = mexikha_war_minimum_commander_martial
					}
				}
			} # Not enough proficient commanders, create more
			# Custom effect utilizes mexikha_war_minimum_commander_martial variable
			log = "SSI.30040 - spawning commanders"
			#mexikha_spawn_commander_effect = yes
			mexikha_spawn_commander_effect = yes
			mexikha_spawn_commander_effect = yes
		}
		
		if = {
			limit = {
				NOT = {
					check_variable = {
						which = mexikha_war_troop_quantity
						value = 1
					}
				}
			}
			break = yes
		}
		
		# Hardcap at 175k troopspawn
		if = {
			limit = {	
				check_variable = {
					which = mexikha_war_troop_quantity
					value > 350
				}
			}
			set_variable = {
				which = mexikha_war_troop_quantity
				value = 350
			}
		}
		
		# Troop spawn override
		if = {
			limit = {
				check_variable = {
					which = mexikha_war_troop_quantity_override
					value > 0
				}
				check_variable = {
					which = mexikha_war_troop_quantity_override
					which = mexikha_war_troop_quantity
				}
			}
			set_variable = {
				which = mexikha_war_troop_quantity
				which = mexikha_war_troop_quantity_override
			}
			set_variable = { 
				which = mexikha_war_troop_quantity_override
				value = 0
			}
		}
		
		845 = { # Anti-Atlas
			save_event_target_as = spawn_province
		}
		if = { #If the NM currently holds land on map
			limit = { is_landed = yes }
			capital_scope = {
				save_event_target_as = spawn_province
			}
		}
		if = { #If the NM is NOT on map
			limit = { is_landed = no }
			persistent_event_target:spawn_province_override = {
				save_event_target_as = spawn_province
			}
		}
		
		set_variable = {
			which = boat_spawns
			which = mexikha_war_troop_quantity
		}
		
		mexikha_war_spawn_troops_effect = yes
		# Requires the current things to function properly:
		# event_target:spawn_province
		# variable: mexikha_war_troop_quality
		# variable: mexikha_war_troop_quantity
		
		multiply_variable = {
			which = boat_spawns
			value = 5
		}
		while = {
			limit = {
				check_variable = {
					which = boat_spawns
					value = 1
				}
			}
			change_variable = {
				which = boat_spawns
				value = -1
			}
			spawn_fleet = {
				province = closest
				owner = ROOT
				disband_on_peace = yes
				troops =
				{
					galleys = { 1 1 }
				}
				merge = yes
			}
		}
			
		FROM = { # Notify enemy of extra troops.
			character_event = { id = SSI.30045 }
		}
	}
}

character_event = {
	id = SSI.30050
	
	hide_window = yes

	has_dlc = "Sunset Invasion"
	
	is_triggered_only = yes
	
	trigger = {
		# on_war_ended_whitepeace
		# Mexikha is a main participant
		offmap_mexikha = {
			governor = {
				OR = {
					character = ROOT
					character = FROM
				}
			}
		}
	}
	
	immediate = {
		character_event = { id = SSI.30053 days = 1 } # Delayed event to clear flags properly

		if = {
			limit = {
				offmap_mexikha = {
					governor = {
						character = ROOT
					}
				}
			}

			### Mexikha was the defender...
			offmap_mexikha = { set_offmap_tmp_flag = mexikha_was_defender_but_white_peace }
			ROOT = { save_event_target_as = the_protector_general }
			FROM = { save_event_target_as = the_aggressor }
		}
		if = {
			limit = {
				offmap_mexikha = {
					governor = {
						character = FROM
					}
				}
			}
			### Mexikha was the aggressor... 	
			offmap_mexikha = { set_offmap_tmp_flag = mexikha_was_aggressor_but_white_peace }
			FROM = { save_event_target_as = the_protector_general }
			ROOT = { save_event_target_as = the_defender }
		}

		if = { limit = { using_unimportant_cb = no } #Makes sure this is not sent during unimportant wars (revolts, etc)...
			any_player = {
				limit = {
					has_offmap_news_enabled = offmap_mexikha
					is_within_diplo_range = event_target:the_protector_general
				}
	    		narrative_event = { id = SSI.10131 days = 1 } #News: "Mexikha white peace'd with someone"
			}
		}
	}
}
